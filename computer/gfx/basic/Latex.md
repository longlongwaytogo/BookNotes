# 矩阵书写测试
----
# 普通矩阵
$$
\begin{matrix}
x & y \\
z & v
\end{matrix}
$$
# 单竖线矩阵
$$
\begin{vmatrix}
	x & y \\
	z & v
	\end{vmatrix}
$$
# 中括号矩阵
$$
\begin{bmatrix}
x & y \\
z & v
\end{bmatrix} 
$$
# 大括号矩阵
$$
\begin{Bmatrix}
x & y \\
z & v
\end{Bmatrix}
$$
# 双竖线矩阵
$$
\begin{Vmatrix}
x & y \\
z & v
\end{Vmatrix}
$$
# 稀疏矩阵

$$
\begin{bmatrix}
0 & \cdots & 0 \\
\vdots & \ddots & \vdots \\
0 & \cdots & 0
\end{bmatrix}
$$
# 圆括号矩阵

$$
\begin{pmatrix}
x & y \\
z & v
\end{pmatrix} 
$$
# 小圆括号矩阵

$$
\bigl( \begin{smallmatrix}
a&b\\

\bigr)
